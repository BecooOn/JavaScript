1.Q: Reversed word
let sentence = prompt("Enter a sentence: ");
let len = sentence.length;
let i = 0; let str = "";
let reverse = ""; let result = "";
for (i = len - 1; i >= 0; i--){
if (sentence[i] !== " ")
{ str += sentence[i]; }
else {
for (let j = str.length - 1; j >= 0; j--)
{ reverse += str[j]; }
result = result + reverse + " ";
str = "";
reverse = "";
}
}
for (let j = str.length - 1; j >= 0; j--) {
reverse += str[j];
}
result += reverse;
console.log(result);
